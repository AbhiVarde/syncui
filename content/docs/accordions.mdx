Sync UI provides several accordion styles with minimal animations and theme-aware designs — ensuring smooth performance, clean visuals, and accessibility.

Each style below is fully responsive, centered, and takes up the full available width.

---

#### Brutalist Style

A bold accordion with sharp edges, solid borders, and subtle motion for a distinctive brutalist aesthetic.

<CodePreview
  preview={<AccordionVariants variant="brutalist" />}
  code={`
import React, { useState } from "react";
import {
  Accordion,
  AccordionSummary,
  AccordionDetails,
  Typography,
  Box,
  useTheme,
} from "@mui/material";
import { motion } from "framer-motion";
import { RxChevronDown } from "react-icons/rx";

const MotionAccordion = motion.create(Accordion);

const BrutalistAccordion = () => {
const theme = useTheme();
const [expandedPanels, setExpandedPanels] = useState({});
const accordionData = [
{ id: "panel1", title: "Getting Started", content: "Quick introduction to our platform with essential setup steps." },
{ id: "panel2", title: "Key Features", content: "Explore powerful tools designed to enhance your workflow." },
{ id: "panel3", title: "Best Practices", content: "Learn proven strategies for optimal performance and results." },
];

const handleChange = (panel) => (\_, isExpanded) => {
setExpandedPanels(prev => ({ ...prev, [panel]: isExpanded }));
};

return (
<Box sx={{ width: "100%", display: "flex", justifyContent: "center" }}>
<Box sx={{ width: "100%", maxWidth: 600 }}>
{accordionData.map((item) => (
<MotionAccordion
key={item.id}
expanded={expandedPanels[item.id] || false}
onChange={handleChange(item.id)}
disableGutters
elevation={0}
sx={{
              backgroundColor: theme.palette.mode === "dark" ? "#111" : "#FFF",
              border: "2px solid",
              borderColor: theme.palette.mode === "dark" ? "#FFF" : "#111",
              boxShadow: theme.palette.mode === "dark" ? "3px 3px 0 #FFF" : "3px 3px 0 #111",
              mb: 2,
              "&::before": { display: "none" },
            }} >
<AccordionSummary
expandIcon={
<motion.div
animate={{ rotate: expandedPanels[item.id] ? 180 : 0 }}
transition={{ duration: 0.2 }}
style={{ color: theme.palette.mode === "dark" ? "#FFF" : "#111" }} >
<RxChevronDown size={20} />
</motion.div>
} >
<Typography variant="h6">{item.title}</Typography>
</AccordionSummary>
<AccordionDetails>
<Typography variant="body2">{item.content}</Typography>
</AccordionDetails>
</MotionAccordion>
))}
</Box>
</Box>
);
};
`}
/>

#### Modern Style

A clean, modern accordion with rounded corners, soft shadows, and simple plus/minus icons for clarity.

<CodePreview
  preview={<AccordionVariants variant="modern" />}
  code={`
import React, { useState } from "react";
import {
  Accordion,
  AccordionSummary,
  AccordionDetails,
  Typography,
  Box,
  useTheme,
  Paper,
} from "@mui/material";
import { motion } from "framer-motion";
import { RxPlus, RxMinus } from "react-icons/rx";

const MotionAccordion = motion.create(Accordion);
const MotionPaper = motion.create(Paper);

const ModernAccordion = () => {
const theme = useTheme();
const [expandedPanels, setExpandedPanels] = useState({});
const accordionData = [
{ id: "panel1", title: "Getting Started", content: "Quick setup steps." },
{ id: "panel2", title: "Key Features", content: "Powerful workflow tools." },
{ id: "panel3", title: "Best Practices", content: "Tips for efficiency." },
];

const handleChange = (panel) => (\_, isExpanded) => {
setExpandedPanels(prev => ({ ...prev, [panel]: isExpanded }));
};

return (
<Box sx={{ width: "100%", display: "flex", justifyContent: "center" }}>
<Box sx={{ width: "100%", maxWidth: 600 }}>
{accordionData.map((item) => (
<MotionPaper
key={item.id}
sx={{
              mb: 2,
              borderRadius: 2,
              border: "1px solid",
              borderColor: theme.palette.mode === "dark" ? "#FFF" : "#111",
              backgroundColor: theme.palette.mode === "dark" ? "#111" : "#FFF",
              color: theme.palette.mode === "dark" ? "#FFF" : "#111",
              boxShadow:
                theme.palette.mode === "dark"
                  ? "0 2px 6px rgba(0,0,0,0.3)"
                  : "0 2px 6px rgba(0,0,0,0.1)",
            }} >
<MotionAccordion
expanded={expandedPanels[item.id] || false}
onChange={handleChange(item.id)}
disableGutters
elevation={0}
sx={{ "&::before": { display: "none" } }} >
<AccordionSummary
expandIcon={
<motion.div
animate={{
                      color: expandedPanels[item.id]
                        ? theme.palette.mode === "dark"
                          ? "#111"
                          : "#FFF"
                        : theme.palette.mode === "dark"
                        ? "#FFF"
                        : "#111",
                      backgroundColor: expandedPanels[item.id]
                        ? theme.palette.mode === "dark"
                          ? "#FFF"
                          : "#111"
                        : "transparent",
                    }}
transition={{ duration: 0.2 }}
style={{
                      borderRadius: "50%",
                      width: 20,
                      height: 20,
                      display: "flex",
                      alignItems: "center",
                      justifyContent: "center",
                    }} >
{expandedPanels[item.id] ? <RxMinus size={20} /> : <RxPlus size={20} />}
</motion.div>
} >
<Typography variant="h6">{item.title}</Typography>
</AccordionSummary>
<AccordionDetails>
<Typography variant="body2">{item.content}</Typography>
</AccordionDetails>
</MotionAccordion>
</MotionPaper>
))}
</Box>
</Box>
);
};
`}
/>

#### Meteor Shower

A card with a dynamic meteor shower animation in the background.

<CodePreview
  preview={<CardVariants variant="meteorShower" />}
  code={`
import React, { useState, useEffect } from "react";
import { Card, CardContent, Typography, Box } from "@mui/material";
import { motion } from "framer-motion";
import { FaMeteor } from "react-icons/fa";

const MeteorShowerCard = () => {
  const [meteors, setMeteors] = useState([]);

  useEffect(() => {
    const createMeteor = () => {
      const newMeteor = {
        id: Math.random(),
        x: Math.random() * 320,
        delay: Math.random() * 2,
      };
      setMeteors((prev) => [...prev, newMeteor]);
      setTimeout(() => {
        setMeteors((prev) => prev.filter((m) => m.id !== newMeteor.id));
      }, 2000);
    };

    const interval = setInterval(createMeteor, 300);
    return () => clearInterval(interval);
  }, []);

  return (
    <Card sx={{ width: 320, height: 200, overflow: "hidden", position: "relative" }}>
      <CardContent sx={{ position: "relative", zIndex: 2 }}>
        <Typography variant="h5" gutterBottom display="flex" alignItems="center">
          <FaMeteor style={{ marginRight: 8, color: "#FF6B6B" }} />
          Meteor Shower
        </Typography>
        <Typography variant="body2">
          Watch the meteors streak across the night sky.
        </Typography>
      </CardContent>
      {meteors.map((meteor) => (
        <Box
          key={meteor.id}
          component={motion.div}
          initial={{ x: meteor.x, y: -20, opacity: 0 }}
          animate={{
            y: 220,
            opacity: [0, 1, 1, 0],
          }}
          transition={{
            duration: 2,
            delay: meteor.delay,
            ease: "linear",
          }}
          sx={{
            position: "absolute",
            width: "2px",
            height: "20px",
            background: "linear-gradient(to bottom, #FFFFFF, transparent)",
            boxShadow: "0 0 10px #FFFFFF",
            zIndex: 1,
          }}
        />
      ))}
    </Card>
  );
};

export default MeteorShowerCard;
`}
/>

#### Multilayer Stack

A stack of cards with a smooth animation effect.

<CodePreview
  preview={<CardVariants variant="multilayerStack" />}
  code={`
import React, { useState, useEffect } from "react";
import { Card, CardContent, Typography, Box } from "@mui/material";
import { motion, AnimatePresence } from "framer-motion";
import { FaLayerGroup } from "react-icons/fa";

const MultilayerStack = () => {
  const [stack, setStack] = useState([0, 1, 2]);
  const [currentIndex, setCurrentIndex] = useState(2);

  useEffect(() => {
    const interval = setInterval(() => {
      setCurrentIndex((prevIndex) => (prevIndex + 1) % stack.length);
    }, 2000);

    return () => clearInterval(interval);
  }, [stack.length]);

  return (
    <Box sx={{ width: 320, height: 200, position: "relative" }}>
      <AnimatePresence>
        {stack.map((item, index) => {
          const isCurrent = index === currentIndex;
          const nextIndex = (currentIndex + 1) % stack.length;
          const isNext = index === nextIndex;

          return (
            <motion.div
              key={item}
              initial={{ opacity: 0, y: 50, scale: 0.8 }}
              animate={{
                y: isCurrent ? 0 : index * 10,
                scale: isCurrent ? 1 : 1 - index * 0.05,
                zIndex: isCurrent ? 4 : 3 - index,
                opacity: isCurrent ? 1 : isNext ? 0.5 : 0,
              }}
              exit={{ opacity: 0, scale: 0.8, y: 50 }}
              transition={{ duration: 0.5, ease: "easeInOut" }}
              style={{
                position: "absolute",
                top: 0,
                left: 0,
                width: "100%",
                height: "100%",
              }}
            >
              <Card sx={{ height: "100%" }}>
                <CardContent>
                  <Typography variant="h5" gutterBottom display="flex" alignItems="center">
                    <FaLayerGroup style={{ marginRight: 8, color: "#2196F3" }} />
                    Stack Card
                  </Typography>
                  <Typography variant="body2">
                    Smooth stack animation in action.
                  </Typography>
                </CardContent>
              </Card>
            </motion.div>
          );
        })}
      </AnimatePresence>
    </Box>
  );
};

export default MultilayerStack;
`}
/>

#### Info Card

An expandable card with smooth animation and interactive content.

<CodePreview
  preview={<CardVariants variant="infoCard" />}
  code={`
import React, { useState } from "react";
import { Card, CardContent, Typography, Box } from "@mui/material";
import { motion, AnimatePresence } from "framer-motion";
import { FaBolt } from "react-icons/fa";
import { RxChevronDown } from "react-icons/rx";

const InfoCard = () => {
  const [isExpanded, setIsExpanded] = useState(false);

  return (
    <motion.div
      layout
      initial={{ borderRadius: 20 }}
      animate={{
        width: isExpanded ? 320 : 250,
        height: isExpanded ? 200 : 150,
      }}
      transition={{
        type: "spring",
        stiffness: 200,
        damping: 20,
        layout: { duration: 0.3 },
      }}
      onClick={() => setIsExpanded(!isExpanded)}
      style={{ cursor: "pointer", overflow: "hidden", position: "relative" }}
    >
      <Card sx={{ height: "100%" }}>
        <CardContent>
          <Typography variant="h5" gutterBottom display="flex" alignItems="center">
            <FaBolt style={{ marginRight: 8, color: "#FFD700" }} />
            Info Card
          </Typography>
          {!isExpanded && (
            <Typography variant="body2" mt={1} display="block">
              Tap to expand for more details.
            </Typography>
          )}
          <AnimatePresence>
            {isExpanded && (
              <motion.div
                initial={{ opacity: 0, y: 20 }}
                animate={{ opacity: 1, y: 0 }}
                exit={{ opacity: 0, y: 20 }}
                transition={{ duration: 0.3 }}
              >
                <Typography variant="body2">
                  Expandable card with smooth animation and interactive content.
                </Typography>
              </motion.div>
            )}
          </AnimatePresence>
        </CardContent>
        <Box
          component={motion.div}
          animate={{
            rotate: isExpanded ? 180 : 0,
            scale: isExpanded ? 1.2 : 1,
          }}
          transition={{ duration: 0.3 }}
          sx={{
            position: "absolute",
            bottom: 16,
            right: 16,
            width: 30,
            height: 30,
            display: "flex",
            alignItems: "center",
            justifyContent: "center",
            borderRadius: "50%",
            backgroundColor: "background.paper",
          }}
        >
          <RxChevronDown size={20} />
        </Box>
      </Card>
    </motion.div>
  );
};

export default InfoCard;
`}
/>

Sync UI offers beautiful, responsive pricing sections to clearly communicate your plans and drive conversions.

#### Three Tier

A clean and modern three-tier pricing table with subtle animations, feature checkmarks, plan icons, and a highlighted primary plan. Ideal for SaaS products with free, pro, and enterprise offerings.

<CodePreview
  noPadding
  preview={<PricingVariants variant="threeTier" />}
  code={`
import React from "react";
import {
  Box,
  Typography,
  Button,
  Container,
  useTheme,
  Stack,
} from "@mui/material";
import { motion } from "motion/react";
import { HugeiconsIcon } from "@hugeicons/react";
import {
  CheckmarkCircle02Icon,
  UserMultipleIcon,
  Rocket02Icon,
  Building02Icon,
  ArrowRight01Icon,
} from "@hugeicons/core-free-icons";

const PricingVariants = ({ variant = "threeTier" }) => {
  const theme = useTheme();
  const isDark = theme.palette.mode === "dark";

  const plans = [
    {
      name: "Starter",
      price: "Free",
      description: "For individuals getting started",
      features: [
        "Core platform access",
        "Custom domain",
        "Web editor",
        "API playground",
      ],
      icon: UserMultipleIcon,
      cta: "Get started",
      primary: false,
    },
    {
      name: "Pro",
      price: "$29",
      suffix: "/month",
      description: "For small teams shipping products",
      features: [
        "Everything in Starter",
        "Team collaboration",
        "Preview deployments",
        "Access control",
      ],
      icon: Rocket02Icon,
      cta: "Start free trial",
      primary: true,
    },
    {
      name: "Enterprise",
      price: "Custom",
      description: "For larger teams with advanced needs",
      features: [
        "Everything in Pro",
        "SSO & permissions",
        "Uptime SLA",
        "Priority support",
      ],
      icon: Building02Icon,
      cta: "Contact sales",
      primary: false,
    },
  ];

  return (
    <Box sx={{ bgcolor: isDark ? "#000" : "#FFF", color: isDark ? "#FFF" : "#000", py: { xs: 8, md: 12 } }}>
      <Container maxWidth="lg">
        <Box textAlign="center" mb={7}>
          <Typography variant="h3" fontWeight={600} mb={1}>Simple pricing</Typography>
          <Typography variant="body1" sx={{ maxWidth: 520, mx: "auto", opacity: 0.65 }}>
            Clear plans designed to scale with your workflow.
          </Typography>
        </Box>

        <Box sx={{ display: "grid", gridTemplateColumns: { xs: "1fr", md: "repeat(3, 1fr)" }, gap: 3 }}>
          {plans.map((plan, index) => (
            <motion.div key={plan.name} initial={{ opacity: 0, y: 20 }} animate={{ opacity: 1, y: 0 }}
              transition={{ duration: 0.35, delay: index * 0.08 }}>
              <Box sx={{
                height: "100%", p: 4, borderRadius: 3, border: "1px solid",
                borderColor: isDark ? "rgba(255,255,255,0.12)" : "rgba(0,0,0,0.12)",
                display: "flex", flexDirection: "column",
                "&:hover": { borderColor: isDark ? "rgba(255,255,255,0.3)" : "rgba(0,0,0,0.3)" },
              }}>
                <Stack spacing={3} flex={1}>
                  <Box textAlign="center">
                    <Box sx={{ display: "inline-flex", p: 1.25, borderRadius: 2, mb: 2,
                      bgcolor: isDark ? "rgba(255,255,255,0.08)" : "rgba(0,0,0,0.06)" }}>
                      <HugeiconsIcon icon={plan.icon} size={24} />
                    </Box>
                    <Typography variant="h6" fontWeight={600}>{plan.name}</Typography>
                  </Box>

                  <Box textAlign="center">
                    <Typography variant="h4" fontWeight={600}>
                      {plan.price}
                      {plan.suffix && <Typography component="span" variant="body2" sx={{ ml: 0.5 }}>{plan.suffix}</Typography>}
                    </Typography>
                    <Typography variant="body2" sx={{ mt: 1, opacity: 0.65 }}>{plan.description}</Typography>
                  </Box>

                  <Stack spacing={1.25}>
                    {plan.features.map((feature) => (
                      <Stack key={feature} direction="row" spacing={1.25} alignItems="center">
                        <HugeiconsIcon icon={CheckmarkCircle02Icon} size={18} />
                        <Typography variant="body2">{feature}</Typography>
                      </Stack>
                    ))}
                  </Stack>

                  <Box mt="auto" pt={3}>
                    <Button fullWidth variant={plan.primary ? "contained" : "text"} sx={{
                      py: 0.6, borderRadius: "999px", fontWeight: 500, textTransform: "none",
                      bgcolor: plan.primary ? (isDark ? "#FFF" : "#000") : "transparent",
                      color: plan.primary ? (isDark ? "#000" : "#FFF") : "inherit",
                      gap: 0.5, "&:hover .arrow": { transform: "translateX(4px)" },
                    }}>
                      {plan.cta}
                      <Box className="arrow" sx={{ transition: "transform 0.2s ease" }}>
                        <HugeiconsIcon icon={ArrowRight01Icon} size={18} />
                      </Box>
                    </Button>
                  </Box>
                </Stack>
              </Box>
            </motion.div>
          ))}
        </Box>
      </Container>
    </Box>
  );
};

export default PricingVariants;
`}
/>

#### Row Layout

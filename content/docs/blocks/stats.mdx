Sync UI offers a variety of stats section styles to showcase key metrics and social proof.

#### Simple

A minimalist row of large numbers with clean typography and staggered animations — great as a divider between sections.

<CodePreview
  noPadding
  preview={<StatsVariants variant="simple" />}
  code={`
import React from "react";
import { Box, Typography, Container, useTheme } from "@mui/material";
import { motion } from "motion/react";

const containerVariants = {
  hidden: {},
  show: {
    transition: {
      staggerChildren: 0.1,
    },
  },
};

const itemVariants = {
  hidden: { opacity: 0, y: 20 },
  show: { opacity: 1, y: 0, transition: { duration: 0.5, ease: "easeOut" } },
};

const SimpleStats = () => {
  const theme = useTheme();
  const isDark = theme.palette.mode === "dark";

  return (
    <Box
      sx={{
        backgroundColor: isDark ? "#000" : "#FFF",
        color: isDark ? "#FFF" : "#000",
        py: { xs: 6, md: 10 },
      }}
    >
      <Container maxWidth="lg">
        <motion.div variants={containerVariants} initial="hidden" animate="show">
          <Box
            sx={{
              display: "grid",
              gridTemplateColumns: { xs: "1fr 1fr", md: "repeat(4, 1fr)" },
              gap: { xs: 4, md: 2 },
              px: { xs: 2, sm: 3 },
            }}
          >
            {[
              { value: "1B+", label: "API requests per day" },
              { value: "99.9%", label: "Uptime guaranteed" },
              { value: "<200ms", label: "Median latency" },
              { value: "180+", label: "Countries served" },
            ].map((stat, i) => (
              <motion.div key={i} variants={itemVariants}>
                <Box sx={{ display: "flex", flexDirection: "column", gap: 0.5 }}>
                  <Typography
                    variant="h3"
                    sx={{ fontWeight: 500, lineHeight: 1, letterSpacing: "-0.02em" }}
                  >
                    {stat.value}
                  </Typography>
                  <Typography
                    variant="body2"
                    sx={{
                      fontWeight: 400,
                      color: isDark ? "rgba(255,255,255,0.5)" : "rgba(0,0,0,0.5)",
                      lineHeight: 1.4,
                    }}
                  >
                    {stat.label}
                  </Typography>
                </Box>
              </motion.div>
            ))}
          </Box>
        </motion.div>
      </Container>
    </Box>
  );
};

export default SimpleStats;
`}
/>

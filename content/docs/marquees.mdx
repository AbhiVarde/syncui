Sync UI provides several marquee animation styles to display dynamic, scrolling content with smooth animations and hover interactions.

#### Horizontal Marquee

A horizontally scrolling carousel with edge gradients and hover pause.

<CodePreview
  preview={<MarqueeVariants variant="horizontal" />}
  code={`
import React, { useState, useEffect } from "react";
import { Box, Card, Stack, Avatar, Typography, IconButton, useTheme } from "@mui/material";
import { motion, useAnimation } from "framer-motion";
import { RxTwitterLogo } from "react-icons/rx";

const componentTestimonials = [
  {
    name: "Emma Rodriguez",
    handle: "@emma_frontend",
    avatar: "https://images.unsplash.com/photo-1494790108377-be9c29b29330",
    text: "SyncUI + Framer Motion = dev time cut in half! ðŸš€",
    verified: true,
  },
  // Add more testimonials like these
]

const HorizontalMarquee = () => {
  const theme = useTheme();
  const controls = useAnimation();
  const [isPaused, setIsPaused] = useState(false);

  useEffect(() => {
    if (!isPaused) {
      controls.start({
        x: "-50%",
        transition: {
          duration: 20,
          ease: "linear",
          repeat: Infinity,
          repeatType: "loop"
        }
      });
    }
    return () => controls.stop();
  }, [isPaused]);

  return (
    <Box sx={{
      position: "relative",
      overflow: "hidden",
      py: 2,
      "&::before, &::after": {
        content: '""',
        position: "absolute",
        top: 0,
        bottom: 0,
        width: 80,
        zIndex: 2,
        background: \`linear-gradient(to \${theme.direction === "rtl" ? "left" : "right"}, \${theme.palette.background.default}00, \${theme.palette.background.default})\`
      },
      "&::after": {
        right: 0,
        background: \`linear-gradient(to \${theme.direction === "rtl" ? "right" : "left"}, \${theme.palette.background.default}00, \${theme.palette.background.default})\`
      }
    }}>
      <motion.div
        animate={controls}
        style={{ display: "flex", gap: theme.spacing(2) }}
        onHoverStart={() => setIsPaused(true)}
        onHoverEnd={() => setIsPaused(false)}
      >
        {[...Array(3)].map((_, i) =>
          componentTestimonials.map((tweet, index) => (
           <Card
              key={\`h-\${i}-\${index}\`}
              sx={{
                minWidth: 280,
                p: 2,
                borderRadius: "12px",
                border: \`1px solid \${theme.palette.divider}\`,
                bgcolor: "background.paper",
                "&:hover": { transform: "translateY(-4px)" }
              }}
            >
              <Stack spacing={1.5}>
                <Stack direction="row" spacing={1.5} alignItems="center">
                  <Avatar
                    src={tweet.avatar}
                    sx={{
                      width: 40,
                      height: 40,
                      border: \`2px solid \${theme.palette.background.paper}\`,
                      boxShadow: theme.shadows[1]
                    }}
                  />
                  <Box>
                    <Typography variant="subtitle1" fontWeight={600}>
                      {tweet.name}
                    </Typography>
                    <Typography variant="body2" color="text.secondary">
                      {tweet.handle}
                    </Typography>
                  </Box>
                  <IconButton size="small" sx={{ color: "#1DA1F2", ml: "auto" }}>
                    <RxTwitterLogo />
                  </IconButton>
                </Stack>
                <Typography variant="body1" sx={{ lineHeight: 1.5 }}>
                  {tweet.text}
                </Typography>
              </Stack>
            </Card>
          ))
        )}
      </motion.div>
    </Box>
  );
};
`}
/>

#### Diagonal Marquee

A diagonal scrolling grid with dynamic rotations and hover effects.

<CodePreview
  preview={<MarqueeVariants variant="diagonal" />}
  code={`
import React, { useState, useEffect } from "react";
import { Box, Card, Stack, Avatar, Typography, IconButton, useTheme } from "@mui/material";
import { motion, useAnimation } from "framer-motion";
import { RxTwitterLogo } from "react-icons/rx";

const componentTestimonials = [
  {
    name: "Emma Rodriguez",
    handle: "@emma_frontend",
    avatar: "https://images.unsplash.com/photo-1494790108377-be9c29b29330",
    text: "SyncUI + Framer Motion = dev time cut in half! ðŸš€",
    verified: true,
  },
  // Add more testimonials like these
]

const DiagonalMarquee = () => {
  const theme = useTheme();
  const controls = useAnimation();
  const [isPaused, setIsPaused] = useState(false);

  useEffect(() => {
    if (!isPaused) {
      controls.start({
        x: "-50%",
        y: "-50%",
        transition: {
          duration: 20,
          ease: "linear",
          repeat: Infinity,
          repeatType: "loop"
        }
      });
    }
    return () => controls.stop();
  }, [isPaused]);

  return (
    <Box sx={{
      height: 400,
      position: "relative",
      overflow: "hidden",
      perspective: 1000
    }}>
      <motion.div
        animate={controls}
        style={{
          display: "grid",
          gridTemplateColumns: "repeat(5, 1fr)",
          gap: theme.spacing(4),
          position: "absolute",
          top: "50%",
          left: "50%",
          transform: "rotate(-8deg)"
        }}
      >
        {[...Array(3)].map((_, i) =>
          componentTestimonials.map((tweet, index) => (
            <motion.div
              key={\`d-\${i}-\${index}\`}
              style={{ transform: \`rotate(\${index % 2 ? 3 : -3}deg)\` }}
              whileHover={{ scale: 1.05 }}
            >
              <Card
              key={\`h-\${i}-\${index}\`}
              sx={{
                minWidth: 280,
                p: 2,
                borderRadius: "12px",
                border: \`1px solid \${theme.palette.divider}\`,
                bgcolor: "background.paper",
                "&:hover": { transform: "translateY(-4px)" }
              }}
            >
              <Stack spacing={1.5}>
                <Stack direction="row" spacing={1.5} alignItems="center">
                  <Avatar
                    src={tweet.avatar}
                    sx={{
                      width: 40,
                      height: 40,
                      border: \`2px solid \${theme.palette.background.paper}\`,
                      boxShadow: theme.shadows[1]
                    }}
                  />
                  <Box>
                    <Typography variant="subtitle1" fontWeight={600}>
                      {tweet.name}
                    </Typography>
                    <Typography variant="body2" color="text.secondary">
                      {tweet.handle}
                    </Typography>
                  </Box>
                  <IconButton size="small" sx={{ color: "#1DA1F2", ml: "auto" }}>
                    <RxTwitterLogo />
                  </IconButton>
                </Stack>
                <Typography variant="body1" sx={{ lineHeight: 1.5 }}>
                  {tweet.text}
                </Typography>
              </Stack>
            </Card>
            </motion.div>
          ))
        )}
      </motion.div>
    </Box>
  );
};
`}
/>

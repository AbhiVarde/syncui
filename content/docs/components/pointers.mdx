#### Glowing Dot

A simple glowing dot that follows the cursor with a subtle glow effect.

<CodePreview
  preview={<PointerVariants variant="glowingDot" />}
  code={`
import React, { useState } from "react";
import { Box, Typography } from "@mui/material";
import { motion } from "motion/react";

const GlowingDot = ({ position }) => {
  return (
    <motion.div
      style={{
        position: "absolute",
        left: position.x,
        top: position.y,
        x: -15,
        y: -15,
        width: 30,
        height: 30,
        borderRadius: "50%",
        background: "rgba(255, 255, 255, 0.8)",
        boxShadow: "0 0 15px 5px rgba(255, 255, 255, 0.5)",
        pointerEvents: "none",
      }}
      initial={{ scale: 0 }}
      animate={{ scale: 1 }}
      transition={{ type: "spring", stiffness: 400, damping: 20 }}
    />
  );
};

const PointerDemo = () => {
  const [isHovering, setIsHovering] = useState(false);
  const [mousePosition, setMousePosition] = useState({ x: 0, y: 0 });

  const handleMouseEnter = () => {
    setIsHovering(true);
  };

  const handleMouseLeave = () => {
    setIsHovering(false);
  };

  const handleMouseMove = (e) => {
    const rect = e.currentTarget.getBoundingClientRect();
    setMousePosition({
      x: e.clientX - rect.left,
      y: e.clientY - rect.top,
    });
  };

  return (
    <Box
      sx={{
        height: "250px",
        width: "300px",
        position: "relative",
        borderRadius: 2,
        overflow: "hidden",
        cursor: "none",
        background: "linear-gradient(135deg, #FF5F6D 0%, #FFC371 100%)",
        display: "flex",
        flexDirection: "column",
        justifyContent: "center",
        alignItems: "center",
        p: 2,
        minHeight: "180px",
      }}
      onMouseEnter={handleMouseEnter}
      onMouseLeave={handleMouseLeave}
      onMouseMove={handleMouseMove}
    >
      <Typography variant="body2" color="white" textAlign="center">
        Move your cursor here
      </Typography>

      {isHovering && <GlowingDot position={mousePosition} />}
    </Box>
  );
};

export default PointerDemo;
`}
/>

#### Following Ring

A ring that follows the cursor with spring physics for smooth movement.

<CodePreview
  preview={<PointerVariants variant="followingRing" />}
  code={`
import React, { useState, useEffect } from "react";
import { Box, Typography } from "@mui/material";
import { motion, useMotionValue, useSpring } from "motion/react";

const FollowingRing = ({ position }) => {
  const springConfig = { stiffness: 250, damping: 30 };
  const x = useMotionValue(position.x);
  const y = useMotionValue(position.y);
  const xSpring = useSpring(x, springConfig);
  const ySpring = useSpring(y, springConfig);

  useEffect(() => {
    x.set(position.x);
    y.set(position.y);
  }, [position, x, y]);

  return (
    <motion.div
      style={{
        position: "absolute",
        left: 0,
        top: 0,
        x: xSpring,
        y: ySpring,
        translateX: "-50%",
        translateY: "-50%",
        width: 40,
        height: 40,
        borderRadius: "50%",
        border: "2px solid white",
        pointerEvents: "none",
      }}
      initial={{ scale: 0 }}
      animate={{ scale: 1 }}

};

#### Trailing Dots

A series of dots that create a trail behind the cursor.

<CodePreview
  preview={<PointerVariants variant="trailingDots" />}
  code={`
import React, { useState, useEffect } from "react";
import { Box, Typography } from "@mui/material";
import { motion } from "motion/react";

const TrailingDots = ({ position }) => {
  const [trail, setTrail] = useState([]);

  useEffect(() => {
    setTrail((prevTrail) => {
      const newTrail = [
        ...prevTrail,
        { x: position.x, y: position.y, id: Date.now() },
      ];
      return newTrail.slice(-5);
    });
  }, [position]);

  return (
    <>
      {trail.map((dot, index) => {
        const size = 12 - index * 2;
        const opacity = 1 - index * 0.2;

        return (
          <motion.div
            key={dot.id}
            style={{
              position: "absolute",
              left: dot.x,
              top: dot.y,
              x: -size / 2,
              y: -size / 2,
              width: size,
              height: size,
              borderRadius: "50%",
              backgroundColor: "white",
              opacity,
              pointerEvents: "none",
            }}
            initial={{ scale: 0 }}
            animate={{ scale: 1 }}
            transition={{ duration: 0.12, ease: "easeOut" }}
          />
        );
      })}
    </>
  );
};

const PointerDemo = () => {
  const [isHovering, setIsHovering] = useState(false);

  const handleMouseEnter = () => {
    setIsHovering(true);

  };

  return (
    <Box

        height: "250px",
        width: "300px",
        position: "relative",
        borderRadius: 2,
        overflow: "hidden",
        cursor: "none",
        background: "linear-gradient(135deg, #9D50BB 0%, #6E48AA 100%)",
        display: "flex",
        flexDirection: "column",
        justifyContent: "center",
        alignItems: "center",
        p: 2,

      onMouseEnter={handleMouseEnter}

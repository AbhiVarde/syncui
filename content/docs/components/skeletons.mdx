Sync UI provides clean skeleton loaders with custom animations built on motion/react — offering shimmer, pulse, and wave effects that feel premium and polished.

Each variant is fully customizable with width, height, and rounded props for complete flexibility.

#### Shimmer

A smooth gradient sweep animation that gives users the "premium app" feel seen in Netflix, Instagram, and LinkedIn.

<CodePreview
  preview={<SkeletonVariants variant="shimmer" width={200} height={20} />}
  code={`
import React from "react";
import { Box, useTheme } from "@mui/material";
import { motion } from "motion/react";

const ShimmerSkeleton = ({ width = "100%", height = 20, rounded = false }) => {
  const theme = useTheme();
  const baseColor = theme.palette.mode === "dark" ? "#2a2a2a" : "#e8e8e8";
  const highlightColor = theme.palette.mode === "dark" ? "#3a3a3a" : "#f5f5f5";

  return (
    <Box
      sx={{
        width,
        height,
        borderRadius: rounded ? 1 : 0,
        overflow: "hidden",
        position: "relative",
        backgroundColor: baseColor,
      }}
    >
      <motion.div
        animate={{ x: ["-100%", "100%"] }}
        transition={{ duration: 1.5, repeat: Infinity, ease: "easeInOut" }}
        style={{
          position: "absolute",
          top: 0,
          left: 0,
          right: 0,
          bottom: 0,
          background: \`linear-gradient(90deg, transparent, \${highlightColor}, transparent)\`,
        }}
      />
    </Box>
  );
};

export default ShimmerSkeleton;
`}
/>

#### Pulse

A subtle breathing effect with smooth opacity transitions — minimal, accessible, and less distracting than shimmer.

<CodePreview
  preview={<SkeletonVariants variant="pulse" width={200} height={20} />}
  code={`
import React from "react";
import { useTheme } from "@mui/material";
import { motion } from "motion/react";

const PulseSkeleton = ({ width = "100%", height = 20, rounded = false }) => {
  const theme = useTheme();
  const baseColor = theme.palette.mode === "dark" ? "#2a2a2a" : "#e8e8e8";

  return (
    <motion.div
      animate={{ opacity: [0.5, 1, 0.5] }}
      transition={{ duration: 1.5, repeat: Infinity, ease: "easeInOut" }}
      style={{
        width,
        height,
        borderRadius: rounded ? "8px" : 0,
        backgroundColor: baseColor,
      }}
    />
  );
};

export default PulseSkeleton;
`}
/>

#### Wave

A unique staggered animation with flowing bars — eye-catching and modern, perfect for dashboard cards and content feeds.

<CodePreview
  preview={<SkeletonVariants variant="wave" width={200} height={20} />}
  code={`
import React from "react";
import { Box, useTheme } from "@mui/material";
import { motion } from "motion/react";

const WaveSkeleton = ({ width = "100%", height = 20, rounded = false }) => {
  const theme = useTheme();
  const baseColor = theme.palette.mode === "dark" ? "#2a2a2a" : "#e8e8e8";

  return (
    <Box
      sx={{
        width,
        height,
        borderRadius: rounded ? 1 : 0,
        overflow: "hidden",
        position: "relative",
      }}
    >
      {[0, 1, 2, 3, 4].map((i) => (
        <motion.div
          key={i}
          animate={{ opacity: [0.3, 1, 0.3] }}
          transition={{
            duration: 1.2,
            repeat: Infinity,
            ease: "easeInOut",
            delay: i * 0.1,
          }}
          style={{
            position: "absolute",
            top: 0,
            left: \`\${i * 20}%\`,
            width: "20%",
            height: "100%",
            backgroundColor: baseColor,
          }}
        />
      ))}
    </Box>
  );
};

export default WaveSkeleton;
`}
/>

### Usage Examples

##### Card Layout

Build a profile card skeleton with avatar and text lines.

```jsx
<Box sx={{ display: "flex", gap: 2, p: 2 }}>
  <SkeletonVariants variant="shimmer" width={60} height={60} rounded />
  <Box sx={{ flex: 1 }}>
    <SkeletonVariants variant="shimmer" width={200} height={20} />
    <Box sx={{ mt: 1 }}>
      <SkeletonVariants variant="shimmer" width={150} height={16} />
    </Box>
  </Box>
</Box>
```

##### Text Lines

Create multi-line text skeletons with varying widths.

```jsx
<Box sx={{ width: "100%" }}>
  <SkeletonVariants variant="pulse" width="100%" height={14} />
  <Box sx={{ mt: 1 }}>
    <SkeletonVariants variant="pulse" width="90%" height={14} />
  </Box>
  <Box sx={{ mt: 1 }}>
    <SkeletonVariants variant="pulse" width="60%" height={14} />
  </Box>
</Box>
```

##### List Items

Build list item skeletons with avatars.

```jsx
<Box sx={{ width: "100%" }}>
  {[1, 2, 3].map((i) => (
    <Box key={i} sx={{ display: "flex", gap: 2, mb: 2, alignItems: "center" }}>
      <SkeletonVariants variant="wave" width={40} height={40} rounded />
      <Box sx={{ flex: 1 }}>
        <SkeletonVariants variant="wave" width="60%" height={14} />
        <Box sx={{ mt: 0.5 }}>
          <SkeletonVariants variant="wave" width="40%" height={10} />
        </Box>
      </Box>
    </Box>
  ))}
</Box>
```

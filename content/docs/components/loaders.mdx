Sync UI offers a variety of loader animation styles to enhance your user interface and provide visual feedback during loading states.

#### Pulsating Dots

Three dots that pulsate in sequence.

<CodePreview
  preview={<LoaderVariants variant="pulsatingDots" />}
  code={`
import React from "react";
import { Box, useTheme } from "@mui/material";
import { motion } from "motion/react";

const PulsatingDots = () => {
  const theme = useTheme();
  const loaderColor = theme.palette.mode === "dark" ? "#FFFFFF" : "#000000";

  return (
    <Box sx={{ height: 40, position: "relative", width: 60 }}>
      {[0, 1, 2].map((i) => (
        <motion.div
          key={i}
          style={{
            width: 12,
            height: 12,
            borderRadius: "50%",
            backgroundColor: loaderColor,
            position: "absolute",
            left: i * 24,
            top: "50%",
            y: "-50%",
          }}
          animate={{
            scale: [1, 1.5, 1],
            opacity: [1, 0.5, 1],
          }}
          transition={{
            duration: 1.2,
            repeat: Infinity,
            delay: i * 0.15,
            ease: "easeInOut",
          }}
        />
      ))}
    </Box>
  );
};

export default PulsatingDots;
`}
/>

#### Bar Wave

  preview={<LoaderVariants variant="barWave" />}
  code={`

const BarWave = () => {
  const theme = useTheme();

  return (
    <Box
      sx={{
        display: "flex",
        justifyContent: "center",
        alignItems: "flex-end",
        height: 40,
        width: 60,
      }}
    >
      {[0, 1, 2, 3, 4].map((i) => (
        <motion.div
          key={i}
          style={{
            width: 4,

            margin: "0 2px",

          animate={{
            height: [10, 30, 10],

            ease: [0.45, 0, 0.55, 1],
            delay: i * 0.08,
          }}
        />
      ))}
    </Box>

export default BarWave;
`}
